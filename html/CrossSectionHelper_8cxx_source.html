276 std::shared_ptr<TH2F> CrossSectionHelper::CrossSection::GetSmearingMatrix(const std::shared_ptr<TH2F> &signalSelectedRecoTrue) const
277 {
278  // Check the input matrix has the correct dimensions
279  const auto nBins = m_binEdges.size() - 1;
280 
281  if (static_cast<unsigned int>(signalSelectedRecoTrue->GetXaxis()->GetNbins()) != nBins ||
282  static_cast<unsigned int>(signalSelectedRecoTrue->GetYaxis()->GetNbins()) != nBins )
283  {
284  throw std::logic_error("CrossSection::GetSmearingMatrix - input matrix has the wrong dimensions");
285  }
286 
287  auto smearingMatrix = this->GetEmptyHist2D();
288  for (unsigned int iTrue = 1; iTrue <= nBins; ++iTrue)
289  {
290  // Get the sum of the reco values for this true bin
291  float summedValue = 0.f;
292  for (unsigned int iReco = 1; iReco <= nBins; ++iReco)
293  {
294  summedValue += signalSelectedRecoTrue->GetBinContent(iTrue, iReco);
295  }
296 
297  if (summedValue <= std::numeric_limits<float>::epsilon())
298  {
299  // ATTN if a true bin has no entries, then we have no information to work with
300  // Here we make the choice to give each reco bin an equal probability
301  const auto value = 1.f / static_cast<float>(nBins);
302  for (unsigned int iReco = 1; iReco <= nBins; ++iReco)
303  {
304  smearingMatrix->SetBinContent(iTrue, iReco, value);
305  }
306  }
307  else
308  {
309  // Normalise the by a constant factor bins such that the columns add to unity
310  const auto norm = 1.f / summedValue;
311 
312  for (unsigned int iReco = 1; iReco <= nBins; ++iReco)
313  {
314  const auto value = signalSelectedRecoTrue->GetBinContent(iTrue, iReco) * norm;
315  smearingMatrix->SetBinContent(iTrue, iReco, value);
316  }
317  }
318  }
319 
320  return smearingMatrix;
321 }

555

557  }

704 CrossSectionHelper::CovarianceBiasPair CrossSectionHelper::CrossSection::GetCrossSectionCovarianceMatrix(const std::string &systParameter)
705 {
706  // Check that we have an entry for the requested systematic paramter
707  const auto iter = m_inputData.m_systUniverseSizesMap.find(systParameter);
708  if (iter == m_inputData.m_systUniverseSizesMap.end())
709  throw std::invalid_argument("CrossSection::GetCrossSectionCovarianceMatrix - unknown parameter: \"" + systParameter + "\"");
710 
711  // Get the number of universes
712  const auto nUniverses = iter->second;
713  if (nUniverses == 0)
714  throw std::logic_error("CrossSection::GetCrossSectionCovarianceMatrix - No universes for parameter: \"" + systParameter + "\"");
715 
716  // Get the nominal cross section
717  const auto crossSectionNom = this->GetCrossSection();
718 
719  // Get the cross-section in each universe and the mean cross-section over all universes
720  std::vector< std::shared_ptr<TH1F> > crossSections;
721 
722  auto nUniversesPerBin = this->GetEmptyHist1D();
723  auto summedCrossSection = this->GetEmptyHist1D();
724  const auto nBins = m_binEdges.size() - 1;
725 
726  //// BEGIN DEBUG
727  std::cout << "Getting covariance matrix for parameter: " << systParameter << std::endl;
728  //// END DEBUG
729 
730  for (unsigned int iUni = 0; iUni < nUniverses; ++iUni)
731  {
732  //// BEGIN DEBUG
733  AnalysisHelper::PrintLoadingBar(iUni, nUniverses);
734  //// END DEBUG
735 
736  // Get the cross-section from the cache (or add it if we haven't calculated this cross section yet)
737  const auto crossSection = this->GetCachedCrossSection(systParameter, iUni);
738  crossSections.push_back(crossSection);
739 
740  // Add this to the sum for each bin
741  for (unsigned int iBin = 1; iBin <= nBins; ++iBin)
742  {
743  if (this->IsUnderOverflowBin(iBin))
744  continue;
745 
746  // Skip dummy bins in which the cross-section couldn't be extracted
747  const auto binValue = crossSection->GetBinContent(iBin);
748  if (std::abs(std::abs(binValue) - std::numeric_limits<float>::max()) < std::numeric_limits<float>::epsilon())
749  continue;
750 
751  summedCrossSection->SetBinContent(iBin, summedCrossSection->GetBinContent(iBin) + binValue);
752  nUniversesPerBin->SetBinContent(iBin, nUniversesPerBin->GetBinContent(iBin) + 1.f);
753  }
754  }
755 
756  // Get the mean cross-section
757  auto meanCrossSection = this->GetEmptyHist1D();
758  for (unsigned int iBin = 1; iBin <= nBins; ++iBin)
759  {
760  if (this->IsUnderOverflowBin(iBin))
761  continue;
762 
763  const auto numerator = summedCrossSection->GetBinContent(iBin);
764  const auto denominator = nUniversesPerBin->GetBinContent(iBin);
765 
766  // If the denominator is zero then use a dummy value
767  const auto isDemoninatorZero = (denominator <= std::numeric_limits<float>::epsilon());
768  const auto mean = isDemoninatorZero ? -std::numeric_limits<float>::max() : (numerator / denominator);
769  meanCrossSection->SetBinContent(iBin, mean);
770  }
771 
772  // Get the covairance matrix and bias vector
773  const auto nAnalysisBins = nBins - (m_hasUnderflow ? 1 : 0) - (m_hasOverflow ? 1 : 0);
774  const std::string covName = "xSecCov_" + std::to_string(m_histCount++);
775  const std::string biasName = "xSecBias_" + std::to_string(m_histCount++);
776  CovarianceBiasPair pair (
777  std::make_shared<TH2F>(covName.c_str(), "", nAnalysisBins, 0, nAnalysisBins, nAnalysisBins, 0, nAnalysisBins),
778  std::make_shared<TH1F>(biasName.c_str(), "", nAnalysisBins, 0, nAnalysisBins)
779  );
780 
781  auto &covarianceMatrix = pair.first;
782  auto &biasVector = pair.second;
783 
784  for (unsigned int iBin = 1; iBin <= nBins; ++iBin)
785  {
786  if (this->IsUnderOverflowBin(iBin))
787  continue;
788 
789  const auto meanI = meanCrossSection->GetBinContent(iBin);
790  const auto nomI = crossSectionNom->GetBinContent(iBin);
791 
792  // Check for dummy bins and warn if they exists
793  if (std::abs(std::abs(nomI) - std::numeric_limits<float>::max()) <= std::numeric_limits<float>::epsilon())
794  {
795  std::cout << "WARNING - Bin: " << iBin << " of nominal cross-section is invalid" << std::endl;
796  continue;
797  }
798 
799  if (std::abs(std::abs(meanI) - std::numeric_limits<float>::max()) <= std::numeric_limits<float>::epsilon())
800  {
801  std::cout << "WARNING - Bin: " << iBin << " of has an incalculable mean value over all universe variations" << std::endl;
802  continue;
803  }
804 
805  // Set the bias vector element
806  const auto bias = meanI - nomI;
807  const auto iBinOutput = iBin - (m_hasUnderflow ? 1 : 0);
808  biasVector->SetBinContent(iBinOutput, bias);
809 
810  for (unsigned int jBin = 1; jBin <= nBins; ++jBin)
811  {
812  if (this->IsUnderOverflowBin(jBin))
813  continue;
814 
815  const auto meanJ = meanCrossSection->GetBinContent(jBin);
816  const auto nomJ = crossSectionNom->GetBinContent(jBin);
817 
818  // Skip dummy bins
819  if (std::abs(std::abs(nomJ) - std::numeric_limits<float>::max()) <= std::numeric_limits<float>::epsilon())
820  continue;
821 
822  if (std::abs(std::abs(meanJ) - std::numeric_limits<float>::max()) <= std::numeric_limits<float>::epsilon())
823  continue;
824 
825  float numerator = 0.f;
826  float denominator = 0.f;
827  for (unsigned int iUni = 0; iUni < nUniverses; ++iUni)
828  {
829  const auto &crossSection = crossSections.at(iUni);
830  const auto sigmaI = crossSection->GetBinContent(iBin);
831  const auto sigmaJ = crossSection->GetBinContent(jBin);
832 
833  // Skip dummy bins
834  if (std::abs(std::abs(sigmaI) - std::numeric_limits<float>::max()) <= std::numeric_limits<float>::epsilon())
835  continue;
836 
837  if (std::abs(std::abs(sigmaJ) - std::numeric_limits<float>::max()) <= std::numeric_limits<float>::epsilon())
838  continue;
839 
840  numerator += (sigmaI - meanI) * (sigmaJ - meanJ);
841  denominator += 1.f;
842  }
843 
844  if (denominator <= std::numeric_limits<float>::epsilon())
845  {
846  throw std::logic_error("CrossSection::GetCrossSectionCovarianceMatrix - For parameter: \"" + systParameter + "\", " +
847  "can't calculate covariance in bin: " + std::to_string(iBin) + ", " + std::to_string(jBin));
848  }
849 
850  // Set the covariance matrix element
851  const auto covariance = numerator / denominator;
852  const auto jBinOutput = jBin - (m_hasUnderflow ? 1 : 0);
853  covarianceMatrix->SetBinContent(iBinOutput, jBinOutput, covariance);
854  }
855  }
856 
857  return pair;
858 }

1507  const auto &covarianceMatrix = covarianceBias.first;
1508 
1509  // Make sure we have square matrices of the same dimensions
1510  const auto &nBins = covarianceMatrix->GetNbinsX();
1511  if (covarianceMatrix->GetNbinsY() != nBins)
1512  throw std::invalid_argument("CrossSectionHelper::GetUncertaintyEigenVectors - Input covariance matrix is not square");
1513 
1514  // Make sure the requested bins are valid
1515  if (iBin == 0 || iBin > static_cast<unsigned int>(nBins))
1516  {
1517  throw std::range_error("CrossSectionHelper::GetUncertaintyEigenVectors - Supplied bin index iBin = " + std::to_string(iBin) +
1518  " is out of range of input matrices: 1 - " + std::to_string(nBins));
1519  }
1520 
1521  if (jBin == 0 || jBin > static_cast<unsigned int>(nBins))
1522  {

1524  " is out of range of input matrices: 1 - " + std::to_string(nBins));
1525  }
1526 
1527  // Get the total error matrix elements for the bins we care about
1528  // Here we take the sum of the covariance and bias matrices as the total error
1529  const auto ii = covarianceMatrix->GetBinContent(iBin, iBin);
1530  const auto ij = covarianceMatrix->GetBinContent(iBin, jBin);
1531  const auto ji = covarianceMatrix->GetBinContent(jBin, iBin);
1532  const auto jj = covarianceMatrix->GetBinContent(jBin, jBin);
1533 
1534  // Get the discriminant of the eigen equation for this 2D covariance matrix
1535  const auto trace = ii + jj;
1536  const auto det = ii*jj - ij*ji;
1537  const auto disc = trace*trace - 4*det;
1538 
1539  if (disc < 0.f)
1540  throw std::logic_error("CrossSectionHelper::GetUncertaintyEigenVectors - Eigen equation has no solutions");
1541 
1542  // Get the eigenvalues
1543  const auto rootDisc = std::pow(disc, 0.5f);
1544  const auto lambdaMax = std::max(trace + rootDisc, trace - rootDisc) / 2.;
1545  const auto lambdaMin = std::min(trace + rootDisc, trace - rootDisc) / 2.;
1546 
1547  // Get the unit eigenvectors
1548  const auto vectMaxUnit = TVector2(ij, lambdaMax - ii).Unit();
1549  const auto vectMinUnit = TVector2(ij, lambdaMin - ii).Unit();
1550 
1551  // Get the eigen vectors which have length of the root of the eigen values
1552  const auto sigmaMax = std::pow(lambdaMax, 0.5f);
1553  const auto sigmaMin = std::pow(lambdaMin, 0.5f);
1554 
1555  const auto vectMax = sigmaMax * vectMaxUnit;
1556  const auto vectMin = sigmaMin * vectMinUnit;
1557 
1558  return std::pair<TVector2, TVector2>(vectMax, vectMin);
1559 }
